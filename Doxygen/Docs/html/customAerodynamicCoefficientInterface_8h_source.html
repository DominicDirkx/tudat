138  virtual void updateCurrentCoefficients( const std::vector< double >& independentVariables )
139  {
140  // Check if the correct number of aerodynamic coefficients is provided.
141  if( independentVariables.size( ) != numberOfIndependentVariables_ )

146  boost::lexical_cast< std::string >( numberOfIndependentVariables_ ) );
147  }
148 
149  // Update current coefficients.
150  Eigen::Vector6d currentCoefficients = coefficientFunction_(
151  independentVariables );
152  currentForceCoefficients_ = currentCoefficients.segment( 0, 3 );
153  currentMomentCoefficients_ = currentCoefficients.segment( 3, 3 );
154  }
