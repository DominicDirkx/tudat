105  boost::shared_ptr< EstimatableParameterSet< InitialStateParameterType > > estimatableParameterSet )
106 {
107  int totalParameterSetSize = estimatableParameterSet->getEstimatedParameterSetSize( );
108  std::vector< boost::shared_ptr< EstimatableParameter< Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > >
109  initialStateParameters = estimatableParameterSet->getEstimatedInitialStateParameters( );
110 
111  for( unsigned int i = 0; i < initialStateParameters.size( ); i++ )
112  {
113  if( ( initialStateParameters.at( i )->getParameterName( ).first != initial_body_state ) )
114  {
115  throw std::runtime_error( "Error when getting single arc paramater vector, did not recognize initial state parameter " +
116  boost::lexical_cast< std::string >( initialStateParameters.at( i )->getParameterName( ).first ) );
117  }
118  }
119  return totalParameterSetSize;
120 }
