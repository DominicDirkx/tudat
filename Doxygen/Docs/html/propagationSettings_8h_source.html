478 template< typename StateScalarType >
479 Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > createCombinedInitialState(
480  const std::map< IntegratedStateType, std::vector< boost::shared_ptr< PropagatorSettings< StateScalarType > > > >&
481  propagatorSettingsList )
482 {
483  // Get total size of propagated state
484  int totalSize = getMultiTypePropagatorStateSize( propagatorSettingsList );
485 
486  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > combinedInitialState =
487  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( totalSize, 1 );
488 
489  // Iterate over all propagation settings and add to total list
490  int currentIndex = 0;
491  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > currentInitialState;
492  for( typename std::map< IntegratedStateType,
493  std::vector< boost::shared_ptr< PropagatorSettings< StateScalarType > > > >::const_iterator
494  typeIterator = propagatorSettingsList.begin( ); typeIterator != propagatorSettingsList.end( ); typeIterator++ )
495  {
496  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
497  {
498  currentInitialState = typeIterator->second.at( i )->getInitialStates( );
499  combinedInitialState.segment( currentIndex, currentInitialState.rows( ) ) = currentInitialState;
500  currentIndex += currentInitialState.rows( );
501  }
502  }
503 
504  return combinedInitialState;
505 }

577  void resetInitialStates( const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& initialBodyStates )
578  {
579  // Iterate over all propagator settings.
580  int currentStartIndex = 0;
581  for( typename std::map< IntegratedStateType,
582  std::vector< boost::shared_ptr< PropagatorSettings< StateScalarType > > > >::iterator
583  propagatorIterator = propagatorSettingsMap_.begin( ); propagatorIterator != propagatorSettingsMap_.end( );
584  propagatorIterator++ )
585  {
586  for( unsigned int i = 0; i < propagatorIterator->second.size( ); i++ )
587  {
588  // Get current state size
589  int currentParameterSize = propagatorIterator->second.at( i )->getInitialStates( ).rows( );
590 
591  // Check consistency
592  if( currentParameterSize + currentStartIndex > initialBodyStates.rows( ) )
593  {
594  throw std::runtime_error(
595  "Error when resetting multi-type state, sizes are incompatible " );
596  }
597 
598  // Reset state for current settings
599  propagatorIterator->second.at( i )->resetInitialStates(
600  initialBodyStates.block( currentStartIndex, 0, currentParameterSize, 1 ) );
601  currentStartIndex += currentParameterSize;
602 
603  }
604  }
605 
606  // Check consistency
607  if( currentStartIndex != initialBodyStates.rows( ) )
608  {
609  std::string errorMessage = "Error when resetting multi-type state, total size is incompatible "+
610  boost::lexical_cast< std::string >( currentStartIndex ) +
611  boost::lexical_cast< std::string >( initialBodyStates.rows( ) );
612  throw std::runtime_error( errorMessage );
613  }
614  }
615

622  std::map< IntegratedStateType, std::vector< boost::shared_ptr< PropagatorSettings< StateScalarType > > > >
623  propagatorSettingsMap_;
624

627 template< typename StateScalarType >
629 
636 std::map< IntegratedStateType, std::vector< std::pair< std::string, std::string > > > getIntegratedTypeAndBodyList(
637  const boost::shared_ptr< PropagatorSettings< StateScalarType > > propagatorSettings )
638 {
639  std::map< IntegratedStateType, std::vector< std::pair< std::string, std::string > > > integratedStateList;
640 
641  // Identify propagator type
642  switch( propagatorSettings->stateType_ )
643  {
644  case hybrid:
645  {
646  boost::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
647  boost::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
648 
649  std::map< IntegratedStateType, std::vector< std::pair< std::string, std::string > > > singleTypeIntegratedStateList;
650 
651 
652  for( typename std::map< IntegratedStateType,
653  std::vector< boost::shared_ptr< PropagatorSettings< StateScalarType > > > >::const_iterator
654  typeIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
655  typeIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); typeIterator++ )
656  {
657  if( typeIterator->first != hybrid )
658  {
659  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
660  {
661  singleTypeIntegratedStateList = getIntegratedTypeAndBodyList< StateScalarType >(
662  typeIterator->second.at( i ) );
663 
664  if( singleTypeIntegratedStateList.begin( )->first != typeIterator->first
665  || singleTypeIntegratedStateList.size( ) != 1 )
666  {
667  std::string errorMessage = "Error when making integrated state list for hybrid propagator, inconsistency encountered " +
668  boost::lexical_cast< std::string >( singleTypeIntegratedStateList.begin( )->first ) + " " +
669  boost::lexical_cast< std::string >( typeIterator->first ) + " " +
670  boost::lexical_cast< std::string >( singleTypeIntegratedStateList.size( ) ) + " " +
671  boost::lexical_cast< std::string >( singleTypeIntegratedStateList.begin( )->second.size( ) );
672  throw std::runtime_error( errorMessage );
673  }
674  else
675  {
676  for( unsigned int j = 0; j < singleTypeIntegratedStateList[ typeIterator->first ].size( ); j++ )
677  {
678  integratedStateList[ typeIterator->first ].push_back(
679  singleTypeIntegratedStateList.begin( )->second.at( j ) );
680  }
681 
682  }
683  }
684  }
685  else
686  {
687  throw std::runtime_error( "Error when making integrated state list, cannot handle hybrid propagator inside hybrid propagator" );
688  }
689  }
690  break;
691  }
692  case transational_state:
693  {
694  boost::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
695  translationalPropagatorSettings = boost::dynamic_pointer_cast<
696  TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
697 
698  if( translationalPropagatorSettings == NULL )
699  {
700  throw std::runtime_error( "Error getting integrated state type list, translational state input inconsistent" );
701  }
702 
703  // Retrieve list of integrated bodies in correct formatting.
704  std::vector< std::pair< std::string, std::string > > integratedBodies;
705  for( unsigned int i = 0; i < translationalPropagatorSettings->bodiesToIntegrate_.size( ); i++ )
706  {
707  integratedBodies.push_back( std::make_pair( translationalPropagatorSettings->bodiesToIntegrate_.at( i ), "" ) );
708  }
709  integratedStateList[ transational_state ] = integratedBodies;
710 
711  break;
712  }
713  case body_mass_state:
714  {
715  boost::shared_ptr< MassPropagatorSettings< StateScalarType > >
716  massPropagatorSettings = boost::dynamic_pointer_cast<
717  MassPropagatorSettings< StateScalarType > >( propagatorSettings );
718  if( massPropagatorSettings == NULL )
719  {
720  throw std::runtime_error( "Error getting integrated state type list, mass state input inconsistent" );
721  }
722 
723  // Retrieve list of integrated bodies in correct formatting.
724  std::vector< std::pair< std::string, std::string > > integratedBodies;
725  for( unsigned int i = 0; i < massPropagatorSettings->bodiesWithMassToPropagate_.size( ); i++ )
726  {
727  integratedBodies.push_back( std::make_pair(
728  massPropagatorSettings->bodiesWithMassToPropagate_.at( i ), "" ) );
729  }
730  integratedStateList[ body_mass_state ] = integratedBodies;
731 
732  break;
733  }
734  case custom_state:
735  {
736  std::vector< std::pair< std::string, std::string > > customList;
737  customList.push_back( std::make_pair( "N/A", "N/A" ) );
738  integratedStateList[ custom_state ] = customList;
739  break;
740  }
741  default:
742  throw std::runtime_error( "Error, could not process integrated state type in getIntegratedTypeAndBodyList " +
743  boost::lexical_cast< std::string >( propagatorSettings->stateType_ ) );
744  }
745 
746  return integratedStateList;
747 }
