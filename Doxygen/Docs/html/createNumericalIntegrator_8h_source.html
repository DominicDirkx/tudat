192 template< typename IndependentVariableType, typename DependentVariableType, typename TimeStepType = IndependentVariableType >
193 boost::shared_ptr< numerical_integrators::NumericalIntegrator< IndependentVariableType, DependentVariableType,
194 DependentVariableType, TimeStepType > > createIntegrator(
195  boost::function< DependentVariableType(
196  const IndependentVariableType, const DependentVariableType& ) > stateDerivativeFunction,
197  const DependentVariableType initialState,
198  boost::shared_ptr< IntegratorSettings< IndependentVariableType > > integratorSettings )
199 
200 {
201  boost::shared_ptr< NumericalIntegrator
202  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > > integrator;
203 
204  // Retrieve requested type of integrator
205  switch( integratorSettings->integratorType_ )
206  {
207  case euler:
208  integrator = boost::make_shared< EulerIntegrator
209  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > >
210  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
211  break;
212  case rungeKutta4:
213 
214  integrator = boost::make_shared< RungeKutta4Integrator
215  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > >
216  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
217  break;
218 
219 
220  case rungeKuttaVariableStepSize:
221  {
222  // Check input consistency
223  boost::shared_ptr< RungeKuttaVariableStepSizeSettings< IndependentVariableType > > variableStepIntegratorSettings =
224  boost::dynamic_pointer_cast< RungeKuttaVariableStepSizeSettings< IndependentVariableType > >(
225  integratorSettings );
226  if( variableStepIntegratorSettings == NULL )
227  {
228  std::runtime_error( "Error, type of integrator settings (rungeKuttaVariableStepSize) not compatible with selected integrator (derived class of IntegratorSettings must be RungeKuttaVariableStepSizeSettings for this type)" );
229  }
230  else
231  {
232  // Get requested RK coefficients and create integrator.
233  RungeKuttaCoefficients coefficients = RungeKuttaCoefficients::get(
234  variableStepIntegratorSettings->coefficientSet_ );
235  integrator = boost::make_shared<
236  RungeKuttaVariableStepSizeIntegrator
237  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > >
238  ( coefficients,
239  stateDerivativeFunction, integratorSettings->initialTime_, initialState,
240  static_cast< TimeStepType >( variableStepIntegratorSettings->minimumStepSize_ ),
241  static_cast< TimeStepType >( variableStepIntegratorSettings->maximumStepSize_ ),
242  variableStepIntegratorSettings->relativeErrorTolerance_,
243  variableStepIntegratorSettings->absoluteErrorTolerance_ ,
244  static_cast< TimeStepType >( variableStepIntegratorSettings->safetyFactorForNextStepSize_ ),
245  static_cast< TimeStepType >( variableStepIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
246  static_cast< TimeStepType >( variableStepIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
247  }
248  break;
249  }
250  default:
251  std::runtime_error(
252  "Error, integrator " + boost::lexical_cast< std::string >( integratorSettings->integratorType_ ) +
253  "not found. " ); }
254  return integrator;
255 }
