581 template< typename StateScalarType = double, typename TimeType = double >
582 boost::shared_ptr< ephemerides::Ephemeris > createTabulatedEphemerisFromSpice(
583  const std::string& body,
584  const TimeType initialTime,
585  const TimeType endTime,
586  const TimeType timeStep,
587  const std::string& observerName,
588  const std::string& referenceFrameName,
589  boost::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
590  boost::make_shared< interpolators::LagrangeInterpolatorSettings >( 8 ) )
591 {
592  using namespace interpolators;
593 
594  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > timeHistoryOfState;
595 
596  // Calculate state from spice at given time intervals and store in timeHistoryOfState.
597  TimeType currentTime = initialTime;
598  while( currentTime < endTime )
599  {
600  timeHistoryOfState[ currentTime ] = spice_interface::getBodyCartesianStateAtEpoch(
601  body, observerName, referenceFrameName, "none", static_cast< double >( currentTime ) ).
602  template cast< StateScalarType >( );
603  currentTime += timeStep;
604  }
605 
606  // Create interpolator.
607  boost::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > > interpolator =
608  interpolators::createOneDimensionalInterpolator(
609  timeHistoryOfState, interpolatorSettings );
610 
611  // Create ephemeris and return.
612  return boost::make_shared< ephemerides::TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
613  interpolator, observerName, referenceFrameName );
614 }
