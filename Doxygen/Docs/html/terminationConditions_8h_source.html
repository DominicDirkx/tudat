42 inline bool checkMaximumIterationsExceeded( const unsigned int numberOfIterations,
43  const unsigned int maximumNumberOfIterations,
44  const bool throwRunTimeException = true )
45 {
46  // Check if maximum number of iterations have been exceeded.
47  const bool isMaximumIterationsExceeded = numberOfIterations > maximumNumberOfIterations;
48 
49  // If exceeded, and flag is set to throw run-time exception, proceed.
50  if ( isMaximumIterationsExceeded && throwRunTimeException )
51  {
52  std::string errorMessage
53  = "Root-finder did not converge within maximum number of iterations!";
54 
55  boost::throw_exception( boost::enable_error_info( std::runtime_error( errorMessage ) ) );
56  }
57 
58  // Else, simply return whether maximum number of iterations have been exceeded.
59  return isMaximumIterationsExceeded;
60 }
