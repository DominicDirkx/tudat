24  AnalyticalSimplePlanetaryTargeter( Eigen::Vector3d departurePosition, Eigen::Vector3d targetPosition,
25  const double travelTime, const double centralBodyGravitationalParameter,
26  const int numberOfRevolutions = 0, const bool rightBranch = true ) : departurePosition_( departurePosition ),
27  targetPosition_( targetPosition ), travelTime_( travelTime ),
28  centralBodyGravitationalParameter_( centralBodyGravitationalParameter ),
29  numberOfRevolutions_(numberOfRevolutions), isRightBranch_( rightBranch )
30  {
31 
32  Eigen::Vector3d H_unnorm = departurePosition_.cross( targetPosition_ );
33  Eigen::Quaterniond quaternions;
34  Eigen::Vector3d e;
35  double theta;
36  Eigen::Vector3d z = Eigen::Vector3d::Zero();
37  z[2] = 1;
38 
39  if( H_unnorm.norm() > 1e-9 )
40  {
41  Eigen::Vector3d h = H_unnorm.normalized();
42  if( h[2] < 0 )
43  h = -h;
44  Eigen::Vector3d h_xy = Eigen::Vector3d::Zero();
45  h_xy.segment(0,2) = h.segment(0,2);
46  e = h.cross( h_xy ).normalized();
47  theta = acos( h.dot( z ) );
48  }
49 
50  else{
51 
52  Eigen::Vector3d E_unnorm = Eigen::Vector3d::Zero();
53  E_unnorm[0] = -departurePosition_[1];
54  E_unnorm[1] = departurePosition_[0];
55  e = E_unnorm.normalized();
56  Eigen::Vector3d departurePosition_xy = Eigen::Vector3d::Zero();
57  departurePosition_xy.segment(0,2) = departurePosition_.segment(0,2);
58  theta = -acos( departurePosition_.normalized().dot( departurePosition_xy.normalized() ) );
59  }
60 
61  quaternions.w() = cos(theta/2);
62  quaternions.x() = e[0]*sin(theta/2);
63  quaternions.y() = e[1]*sin(theta/2);
64  quaternions.z() = e[2]*sin(theta/2);
65 
66 
67  Eigen::Matrix3d rotationM = quaternions.normalized().toRotationMatrix();
68  Eigen::Vector3d inPlaneDeparturePosition = rotationM.inverse() * departurePosition_;
69  Eigen::Vector3d inPlaneTargetPosition = rotationM.inverse() * targetPosition_;
70  inPlaneDeparturePosition[2] = 0.0;
71  inPlaneTargetPosition[2] = 0.0;
72  mission_segments::MultiRevolutionLambertTargeterIzzo lambertTargeter(
73  inPlaneDeparturePosition, inPlaneTargetPosition,
74  travelTime_, centralBodyGravitationalParameter_ );
75 
76  const int maxNumberOfRevolutions = lambertTargeter.getMaximumNumberOfRevolutions();
77 
78  if( numberOfRevolutions_ > maxNumberOfRevolutions )
79  {
80  throw std::runtime_error( "The specified number of revolutions is larger than the maximum" );
81  }
82 
83  lambertTargeter.computeForRevolutionsAndBranch( numberOfRevolutions_, isRightBranch_ );
84 
85  departureVelocity_ = rotationM * lambertTargeter.getInertialVelocityAtDeparture();
86 
87  arrivalVelocity_ = rotationM * lambertTargeter.getInertialVelocityAtArrival();
88 
89  }
90 
91  Eigen::Vector3d departurePosition_;
92  Eigen::Vector3d targetPosition_;
93  double travelTime_;
94  double centralBodyGravitationalParameter_;
95  int numberOfRevolutions_;
96  bool isRightBranch_;
97 
98  Eigen::Vector3d departureVelocity_;
99  Eigen::Vector3d arrivalVelocity_;
100 
101 };
