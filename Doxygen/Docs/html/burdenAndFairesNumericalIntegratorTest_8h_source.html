61  std::pair< double, bool > computeNewStepSize( const double stepSize, const double lowerOrder,
62  const double higherOrder,
63  const double safetyFactorForNextStepSize,
64  const Eigen::VectorXd& relativeErrorTolerance,
65  const Eigen::VectorXd& absoluteErrorTolerance,
66  const Eigen::VectorXd& lowerOrderEstimate,
67  const Eigen::VectorXd& higherOrderEstimate )
68  {
69  TUDAT_UNUSED_PARAMETER( higherOrder );
70  TUDAT_UNUSED_PARAMETER( relativeErrorTolerance );
71 
72  // Set higher and lower order estimates.
73  higherOrderEstimate_ = higherOrderEstimate;
74  lowerOrderEstimate_ = lowerOrderEstimate;
75 
76  // Compute truncation error estimate based on higher and lower estimates.
77  const Eigen::VectorXd trucationError = ( higherOrderEstimate
78  - lowerOrderEstimate ).array( ).abs( );
79 
80  // Compute relative error.
81  relativeError_ = trucationError.array( ).abs( ) / stepSize;
82 
83  // Compute new step size for next integration step. (In case the step size for this
84  // current integration step doesn't satisfy the tolerances set, this new step size is
85  // used to recompute the current step.
86  const double newStepSize = safetyFactorForNextStepSize * stepSize * std::pow(
87  absoluteErrorTolerance.array( ).minCoeff( )
88  / relativeError_.array( ).abs( ).maxCoeff( ), 1.0 / lowerOrder );
89 
90  // Check if the current integration step is accepted, based on the allowed absolute
91  // tolerance.
92  const bool isIntegrationStepAccepted = relativeError_.array( ).abs( ).maxCoeff( )
93  < absoluteErrorTolerance.array( ).minCoeff( );
94 
95  // Return the computed new step size and a flag whether the current step is accepted or
96  // not.
97  return std::make_pair( newStepSize, isIntegrationStepAccepted );
98  }
