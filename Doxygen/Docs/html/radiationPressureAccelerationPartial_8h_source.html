198  void update( const double currentTime = 0.0 )
199  {
200  if( !( currentTime_ == currentTime ) )
201  {
202  // Compute helper quantities.
203  Eigen::Vector3d rangeVector = ( acceleratedBodyState_( ) - sourceBodyState_( ) );
204  double range = rangeVector.norm( );
205  double rangeInverse = 1.0 / ( range );
206 
207  // Compute position partial.
208  currentPartialWrtPosition_ =
209  ( radiationPressureCoefficientFunction_( ) * areaFunction_( ) * radiationPressureFunction_( ) /
210  acceleratedBodyMassFunction_( ) ) * ( Eigen::Matrix3d::Identity( ) * rangeInverse - 3.0 *
211  rangeVector * rangeVector.transpose( ) * rangeInverse / (
212  range * range ) );
213  currentTime_ = currentTime;
214  }
