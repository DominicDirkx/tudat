185 template< typename InputIterator >
186 void writeDataMapToTextFile(
187  InputIterator iteratorDataMap, InputIterator last,
188  const std::string& outputFilename,
189  const boost::filesystem::path& outputDirectory, const std::string& fileHeader,
190  const int precisionOfKeyType, const int precisionOfValueType,
191  const std::string& delimiter )
192 {
193  // Check if output directory exists; create it if it doesn't.
194  if ( !boost::filesystem::exists( outputDirectory ) )
195  {
196  boost::filesystem::create_directories( outputDirectory );
197  }
198 
199  // Open output file.
200  std::string outputDirectoryAndFilename = outputDirectory.string( ) + "/" + outputFilename;
201  std::ofstream outputFile_( outputDirectoryAndFilename.c_str( ) );
202 
203  // Write file header to file.
204  outputFile_ << fileHeader;
205 
206  // Loop over map of propagation history.
207  for ( ; iteratorDataMap != last; iteratorDataMap++ )
208  {
209  // Print map data to output file.
210  outputFile_ << std::setprecision( precisionOfKeyType )
211  << std::left << std::setw( precisionOfKeyType + 1 )
212  << iteratorDataMap->first;
213  writeValueToStream( outputFile_, iteratorDataMap->second, precisionOfValueType,
214  delimiter );
215  }
216 
217  // Close output file.
218  outputFile_.close( );
219 }
