116 template< typename DataType >
117 DataType extractParameterValue( const DataLineIterator& firstDataLine,
118  const DataLineIterator& lastDataLine,
119  const DictionaryIterator& dictionaryEntry,
120  const DataType& defaultValue = DataType( ),
121  const boost::function< DataType( DataType ) >& convert
122  = &convertDummy< DataType > )
123 {
124  // Attempt to match dictionary entry with any data lines.
125  DataLineIterator parsedDataVectorIterator
126  = std::find_if( firstDataLine, lastDataLine, DictionaryComparer( *dictionaryEntry ) );
127 
128  // If the data line found is the last data line (typically iterator to the end of the
129  // container), then check if the dictionary entry is required.
130  if ( parsedDataVectorIterator == lastDataLine )
131  {
132  // If the entry is required, throw an error indicating that the parameter cannot be found
133  // in the input stream.
134  if ( ( *dictionaryEntry )->isRequired )
135  {
136  boost::throw_exception(
137  boost::enable_error_info(
138  std::runtime_error(
139  "Required parameter \"" + ( *dictionaryEntry )->parameterName
140  + "\" not found in input stream! " ) ) );
141  }
142 
143  // Else, return the default value specified.
144  else
145  {
146  return defaultValue;
147  }
148  }
149 
150  // Else, set the flag in the dictionary entry indicating that it is extracted to true and
151  // return the extract value using the getField() function.
152  else
153  {
154  ( *dictionaryEntry )->isExtracted = true;
155 
156  return convert( parsed_data_vector_utilities::getField< DataType >(
157  *parsedDataVectorIterator, field_types::general::parameterValue ) );
158  }
159 
160  // Dummy return to ensure no warnings are output.
161  return DataType( );
162 }
