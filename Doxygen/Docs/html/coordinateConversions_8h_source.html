66 template< typename ScalarType = double >
67 Eigen::Matrix< ScalarType, 3, 1 > convertCartesianToSpherical( const Eigen::Matrix< ScalarType, 3, 1 > & cartesianCoordinates )
68 
69 {
70  // Create output Vector3d.
71  Eigen::Matrix< ScalarType, 3, 1 > convertedSphericalCoordinates_ = Eigen::Matrix< ScalarType, 3, 1 >::Zero( );
72 
73  // Compute transformation of Cartesian coordinates to spherical coordinates.
74  convertedSphericalCoordinates_( 0 ) = cartesianCoordinates.norm( );
75 
76  // Check if coordinates are at origin.
77  if ( convertedSphericalCoordinates_( 0 ) < std::numeric_limits< ScalarType >::epsilon( ) )
78  {
79  convertedSphericalCoordinates_( 1 ) = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
80  convertedSphericalCoordinates_( 2 ) = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
81  }
82  // Else compute coordinates using trigonometric relationships.
83  else
84  {
85  convertedSphericalCoordinates_( 1 ) = std::acos( cartesianCoordinates( 2 )
86  / convertedSphericalCoordinates_( 0 ) );
87  convertedSphericalCoordinates_( 2 ) = std::atan2( cartesianCoordinates( 1 ),
88  cartesianCoordinates( 0 ) );
89  }
90 
91  return convertedSphericalCoordinates_;
92 }
