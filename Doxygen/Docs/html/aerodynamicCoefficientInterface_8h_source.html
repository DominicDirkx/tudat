138  void updateCurrentControlSurfaceCoefficientsCoefficients(
139  const std::string& currentControlSurface,
140  std::vector< double > controlSurfaceIndependentVariables )
141  {
142  if( controlSurfaceIncrementInterfaces_.count( currentControlSurface ) == 0 )
143  {
144  throw std::runtime_error( "Error when updating coefficients, could not fid control surface " + currentControlSurface );
145  }
146  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->updateCurrentCoefficients(
147  controlSurfaceIndependentVariables );
148  currentForceCoefficients_ +=
149  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentForceCoefficients( );
150  currentMomentCoefficients_ +=
151  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentMomentCoefficients( );
152  }
