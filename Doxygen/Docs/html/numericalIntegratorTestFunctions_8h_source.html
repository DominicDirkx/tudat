72 static inline Eigen::VectorXd computeVanDerPolStateDerivative( const double time,
73  const Eigen::VectorXd& state )
74 {
75  // Declare van der Pol state derivative vector.
76  Eigen::VectorXd vanDerPolStateDerivative( state.rows( ) );
77 
78  // Compute state derivative.
79  vanDerPolStateDerivative( 0 ) = state( 1 );
80  vanDerPolStateDerivative( 1 ) = ( 1.0 - std::pow( state( 0 ), 2.0 ) ) * state( 1 )
81  + state( 0 );

83  // Return computed state derivative.
84  return vanDerPolStateDerivative;
