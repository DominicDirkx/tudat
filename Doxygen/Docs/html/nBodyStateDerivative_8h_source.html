88  NBodyStateDerivative( const basic_astrodynamics::AccelerationMap& accelerationModelsPerBody,
89  const boost::shared_ptr< CentralBodyData< StateScalarType, TimeType > > centralBodyData,
90  const TranslationalPropagatorType propagatorType,
91  const std::vector< std::string >& bodiesToIntegrate ):
92  propagators::SingleStateTypeDerivative< StateScalarType, TimeType >(
93  propagators::transational_state ),
94  accelerationModelsPerBody_( accelerationModelsPerBody ),
95  centralBodyData_( centralBodyData ),
96  propagatorType_( propagatorType ),
97  bodiesToBeIntegratedNumerically_( bodiesToIntegrate )
98  {
99  // Add empty acceleration map if body is to be propagated with no accelerations.
100  for( unsigned int i = 0; i < bodiesToBeIntegratedNumerically_.size( ); i++ )
101  {
102  if( accelerationModelsPerBody_.count( bodiesToBeIntegratedNumerically_.at( i ) ) == 0 )
103  {
104  accelerationModelsPerBody_[ bodiesToBeIntegratedNumerically_.at( i ) ] =
105  basic_astrodynamics::SingleBodyAccelerationMap( );
106  }
107  }
108 
109  // Correct order of propagated bodies.
110  for( outerAccelerationIterator = accelerationModelsPerBody_.begin( );
111  outerAccelerationIterator != accelerationModelsPerBody_.end( );
112  outerAccelerationIterator++ )
113  {
114  std::vector< std::string >::iterator findIterator =
115  std::find( bodiesToBeIntegratedNumerically_.begin( ), bodiesToBeIntegratedNumerically_.end( ),
116  outerAccelerationIterator->first );
117  bodyOrder_.push_back( std::distance( bodiesToBeIntegratedNumerically_.begin( ), findIterator ) );
118  }
119 
120  createAccelerationModelList( );
121  }

260  Eigen::Vector3d getTotalAccelerationForBody(
261  const std::string& bodyName )
262  {
263  // Check if body is propagated.
264  Eigen::Vector3d totalAcceleration = Eigen::Vector3d::Zero( );
265  if( std::find( bodiesToBeIntegratedNumerically_.begin( ),
266  bodiesToBeIntegratedNumerically_.end( ),
267  bodyName ) == bodiesToBeIntegratedNumerically_.end( ) )
268  {
269  std::string errorMessage = "Error when getting total acceleration for body " + bodyName +
270  ", no such acceleration is found";
271  throw std::runtime_error( errorMessage );
272  }
273  else
274  {
275  if( accelerationModelsPerBody_.count( bodyName ) != 0 )
276  {
277  basic_astrodynamics::SingleBodyAccelerationMap accelerationsOnBody =
278  accelerationModelsPerBody_.at( bodyName );
279 
280  // Iterate over all accelerations acting on body
281  for( innerAccelerationIterator = accelerationsOnBody.begin( );
282  innerAccelerationIterator != accelerationsOnBody.end( );
283  innerAccelerationIterator++ )
284  {
285  for( unsigned int j = 0; j < innerAccelerationIterator->second.size( ); j++ )
286  {
287  // Calculate acceleration and add to state derivative.
288  totalAcceleration += innerAccelerationIterator->second[ j ]->getAcceleration( );
289  }
290  }
291  }
292  }
293  return totalAcceleration;
294  }

340  void sumStateDerivativeContributions(
341  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
342  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
343  {
344  using namespace basic_astrodynamics;
345 
346  stateDerivative.setZero( );
347 
348  int currentBodyIndex = 0;
349  int currentAccelerationIndex = 0;
350 
351  // Iterate over all bodies with accelerations.
352  for( outerAccelerationIterator = accelerationModelsPerBody_.begin( );
353  outerAccelerationIterator != accelerationModelsPerBody_.end( );
354  outerAccelerationIterator++ )
355  {
356  currentBodyIndex = bodyOrder_[ currentAccelerationIndex ];
357 
358  // Iterate over all accelerations acting on body
359  for( innerAccelerationIterator = outerAccelerationIterator->second.begin( );
360  innerAccelerationIterator != outerAccelerationIterator->second.end( );
361  innerAccelerationIterator++ )
362  {
363  for( unsigned int j = 0; j < innerAccelerationIterator->second.size( ); j++ )
364  {
365  // Calculate acceleration and add to state derivative.
366  stateDerivative.block( currentBodyIndex * 6 + 3, 0, 3, 1 ) += (
367  innerAccelerationIterator->second[ j ]->getAcceleration( ) ).
368  template cast< StateScalarType >( );
369  }
370  }
371 
372  // Add body velocity as derivative of its position.
373  stateDerivative.block( currentBodyIndex * 6, 0, 3, 1 ) =
374  ( stateOfSystemToBeIntegrated.segment( currentBodyIndex * 6 + 3, 3 ) );
375  currentAccelerationIndex++;
376  }
377  }
