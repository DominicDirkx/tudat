33 struct TestFunctionWithLargeRootDifference : public TestFunction,
34  public basic_mathematics::BasicFunction< double, double >

37  unsigned int maximumDerivativeOrder;
38 
40  TestFunctionWithLargeRootDifference( const unsigned int aMaximumDerivativeOrder,
41  const double anIncomingSemiMajorAxis,
42  const double anOutgoingSemiMajorAxis,
43  const double aBendingAngle )
44  : maximumDerivativeOrder( aMaximumDerivativeOrder ),
45  incomingSemiMajorAxis( anIncomingSemiMajorAxis ),
46  outgoingSemiMajorAxis( anOutgoingSemiMajorAxis ),
47  bendingAngle( aBendingAngle )
48  { }
49 
51  double evaluate( const double inputValue )
52  {
53  return std::asin( 1.0 / inputValue )
54  + std::asin( 1.0 / ( 1.0 - incomingSemiMajorAxis / outgoingSemiMajorAxis *
55  ( 1.0 - inputValue ) ) ) - bendingAngle;
56  }
57 
59  double computeDerivative( const unsigned int order, const double inputValue )
60  {
61  // Sanity check.
62  if ( order > maximumDerivativeOrder )
63  {
64  throw std::runtime_error( "The rootfinder should not evaluate higher derivatives!" );
65  }
66 
67  // Return the analytical expression for the derivatives.
68  if ( order == 0 )
69  {
70  // Return the function value.
71  return evaluate( inputValue );
72  }
73 
74  else if ( order == 1 )
75  {
76  // Return the first derivative function value.
77  const double eccentricitySquareMinusOne = inputValue * inputValue - 1.0;
78  const double semiMajorAxisRatio = incomingSemiMajorAxis / outgoingSemiMajorAxis;
79  const double bParameter = 1.0 - semiMajorAxisRatio * ( 1.0 - inputValue );
80 
81  return -1.0 / ( inputValue * std::sqrt( eccentricitySquareMinusOne ) ) -
82  semiMajorAxisRatio / ( bParameter * std::sqrt( bParameter * bParameter - 1.0 ) );
83  }
84 
85  else
86  {
87  throw std::runtime_error(
88  "An error occured when evaluating the order of the derivative." );
89  }
90  }

93  double computeDefiniteIntegral( unsigned int order, double lowerBound, double upperbound )
