106  virtual StateType performIntegrationStep( const TimeStepType stepSize )
107  {
108  lastIndependentVariable_ = currentIndependentVariable_;
109  lastState_ = currentState_;
110 
111  // Calculate k1-k4.
112  const StateDerivativeType k1 = stepSize * this->stateDerivativeFunction_(
113  currentIndependentVariable_, currentState_ );
114 
115  const StateDerivativeType k2 = stepSize * this->stateDerivativeFunction_(
116  currentIndependentVariable_ + stepSize / 2.0,
117  static_cast< StateType >( currentState_ + k1 / 2.0 ) );
118 
119  const StateDerivativeType k3 = stepSize * this->stateDerivativeFunction_(
120  currentIndependentVariable_ + stepSize / 2.0,
121  static_cast< StateType >( currentState_ + k2 / 2.0 ) );
122 
123  const StateDerivativeType k4 = stepSize * this->stateDerivativeFunction_(
124  currentIndependentVariable_ + stepSize,
125  static_cast< StateType >( currentState_ + k3 ) );
126 
127  stepSize_ = stepSize;
128  currentIndependentVariable_ += stepSize_;
129  currentState_ += ( k1 + 2.0 * k2 + 2.0 * k3 + k4 ) / 6.0;
130 
131  // Return the integration result.
132  return currentState_;
133  }
