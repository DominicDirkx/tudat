151 template < typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
152 StateType NumericalIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >::integrateTo(
153  const IndependentVariableType intervalEnd,
154  const TimeStepType initialStepSize,
155  const TimeStepType finalTimeTolerance )
156 {
157  TimeStepType stepSize = initialStepSize;
158 
159  // Flag to indicate that the integration end value of the independent variable has been
160  // reached.
161  bool atIntegrationIntervalEnd = static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) )
162  * stepSize / std::fabs( stepSize )
163  <= finalTimeTolerance;
164 
165  int loopCounter = 0;
166  while ( !atIntegrationIntervalEnd )
167  {
168  // Check if the remaining interval is smaller than the step size.
169  if ( std::fabs( static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) ) )
170  <= std::fabs( stepSize ) *
171  ( 1.0 + finalTimeTolerance ) )
172  {
173  // The next step is beyond the end of the integration interval, so adjust the
174  // step size accordingly.
175  stepSize = intervalEnd - getCurrentIndependentVariable( );
176 
177  // Explicitly flag that the integration interval end is reached. Due to rounding
178  // off errors, it may not be possible to use
179  // ( currentIndependentVariable >= independentVariableEnd ) // in the while condition.
180  atIntegrationIntervalEnd = true;
181 
182  }
183 
184  // Perform the step.
185  performIntegrationStep( stepSize );
186  stepSize = getNextStepSize( );
187 
188  // Only applicable to adaptive step size methods:
189  // Perform additional step(s) to reach intervalEnd exactly, in case the last step was rejected by
190  // the variable step size routine and a new step was used that was too small to get to intervalEnd.
191  if ( atIntegrationIntervalEnd )
192  {
193  // As long as intervalEnd is not reached, perform additional steps with the remaining time
194  // as suggested step size for the variable step size routine.
195  if( std::fabs( static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) ) ) >
196  finalTimeTolerance )
197  {
198  // Ensure that integrateTo function does not get stuck in a loop.
199  if( loopCounter < 1000 )
200  {
201  atIntegrationIntervalEnd = false;
202  loopCounter++;
203  }
204  else
205  {
206  std::cerr<<"Warning, integrateTo function has failed to converge to final time to within tolerances, difference between true and requested final time is "<<
207  intervalEnd - getCurrentIndependentVariable( ) <<", final time is: "<<
208  getCurrentIndependentVariable( )<<std::endl;
209  }
210  }
211  }
212  }
213 
214  return getCurrentState( );
215 }
