70 template< typename StateScalarType >
71 Eigen::Matrix< StateScalarType, 6, 1 > transformStateToFrameFromRotations(
72  const Eigen::Matrix< StateScalarType, 6, 1 >& stateInBaseFrame,
73  const Eigen::Quaterniond& rotationToFrame,
74  const Eigen::Matrix3d& rotationMatrixToFrameDerivative )
75 {
76  Eigen::Matrix< StateScalarType, 6, 1 >stateInTargetFrame;
77  stateInTargetFrame.segment( 0, 3 ) =
78  rotationToFrame.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 0, 3 );
79  stateInTargetFrame.segment( 3, 3 ) =
80  rotationMatrixToFrameDerivative.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 0, 3 ) +
81  rotationToFrame.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 3, 3 );
82  return stateInTargetFrame;
83 
84 }
