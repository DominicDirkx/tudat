248  {
249  Eigen::Matrix< ParameterScalar, Eigen::Dynamic, 1 > parameterValues =
250  Eigen::Matrix< ParameterScalar, Eigen::Dynamic, 1 >::Zero( totalParameterSetSize_ );
251 
252  int currentStartIndex = 0;
253 
254  // Retrieve initial state parameter values.
255  for( unsigned int i = 0; i < estimateInitialStateParameters_.size( ); i++ )
256  {
257  parameterValues.segment( currentStartIndex, estimateInitialStateParameters_[ i ]->getParameterSize( ) ) =
258  estimateInitialStateParameters_[ i ]->getParameterValue( ).template cast< ParameterScalar >( );
259  currentStartIndex += estimateInitialStateParameters_[ i ]->getParameterSize( );
260  }
261 
262  // Retrieve double parameter values.
263  for( unsigned int i = 0; i < estimatedDoubleParameters_.size( ); i++ )
264  {
265  parameterValues( currentStartIndex ) = static_cast< ParameterScalar >(
266  estimatedDoubleParameters_[ i ]->getParameterValue( ) );
267  currentStartIndex++;
268  }
269 
270  // Retrieve vector parameter values.
271  for( unsigned int i = 0; i < estimatedVectorParameters_.size( ); i++ )
272  {
273  parameterValues.segment( currentStartIndex, estimatedVectorParameters_[ i ]->getParameterSize( ) ) =
274  estimatedVectorParameters_[ i ]->getParameterValue( ).template cast< ParameterScalar >( );
275  currentStartIndex += estimatedVectorParameters_[ i ]->getParameterSize( );
276  }

286  template< typename ParameterScalar >
287  void resetParameterValues( const Eigen::Matrix< ParameterScalar, Eigen::Dynamic, 1 >& newParameterValues )
288  {
289  // Check input consistency
290  if( newParameterValues.rows( ) != totalParameterSetSize_ )
291  {
292  throw std::runtime_error( "Error when resetting parameters of parameter set, given vector has size " +
293  boost::lexical_cast< std::string >( newParameterValues.rows( ) ) +
294  ", while internal size is " + boost::lexical_cast< std::string >( totalParameterSetSize_ ) );
295  }
296  else
297  {
298  int currentStartIndex = 0;
299 
300  for( unsigned int i = 0; i < estimateInitialStateParameters_.size( ); i++ )
301  {
302  estimateInitialStateParameters_[ i ]->setParameterValue(
303  newParameterValues.segment( currentStartIndex, estimateInitialStateParameters_[ i ]->getParameterSize( ) ).
304  template cast< InitialStateParameterType >( ) );
305  currentStartIndex += estimateInitialStateParameters_[ i ]->getParameterSize( );
306  }
307 
308  // Set double parameter values.
309  for( unsigned int i = 0; i < estimatedDoubleParameters_.size( ); i++ )
310  {
311  estimatedDoubleParameters_[ i ]->setParameterValue( static_cast< double >( newParameterValues( currentStartIndex ) ) );
312  currentStartIndex++;
313  }
314 
315  // Set vector parameter values.
316 
317  for( unsigned int i = 0; i < estimatedVectorParameters_.size( ); i++ )
318  {
319  estimatedVectorParameters_[ i ]->setParameterValue(
320  newParameterValues.segment( currentStartIndex, estimatedVectorParameters_[ i ]->getParameterSize( ) ).
321  template cast< double >( ) );
322 
323  currentStartIndex += estimatedVectorParameters_[ i ]->getParameterSize( );
324  }
325  }
326  }

510 template< typename InitialStateParameterType = double >
511 Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > getInitialStateVectorOfBodiesToEstimate(
512  const boost::shared_ptr< EstimatableParameterSet< InitialStateParameterType > > estimatableParameters )
513 {
514  // Retrieve initial dynamical parameters.
515  std::vector< boost::shared_ptr< EstimatableParameter<
516  Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
517  estimatableParameters->getEstimatedInitialStateParameters( );
518 
519  // Initialize state vector.
520  Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > initialStateVector =
521  Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 >::Zero(
522  estimatableParameters->getInitialDynamicalStateParameterSize( ), 1 );
523 
524  int vectorSize = 0;
525  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
526  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
527  {
528  if( isParameterDynamicalPropertyInitialState( initialDynamicalParameters.at( i )->getParameterName( ).first ) )
529  {
530  int currentParameterSize = initialDynamicalParameters.at( i )->getParameterSize( );
531  initialStateVector.block( vectorSize, 0, currentParameterSize, 1 ) = initialDynamicalParameters.at( i )->getParameterValue( );
532 
533  vectorSize += currentParameterSize;
534  }
535  }
536 
537  return initialStateVector.block( 0, 0, vectorSize, 1 );
538 }
