96  Eigen::MatrixXd wrtParameter(
97  const boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
98  {
99  // Initialize partial
100  Eigen::MatrixXd partial = Eigen::MatrixXd( accelerationSize_, 1 );
101 
102  // Get partial computation function.
103  std::pair< boost::function< void( Eigen::MatrixXd& ) >, int > partialFunction =
104  getParameterPartialFunction( parameter );
105 
106  // If parameter dependency exists, compute it, otherwise set partial to zero.
107  if( partialFunction.second > 0 )
108  {
109  partialFunction.first( partial );
110  }
111  else
112  {
113  partial = Eigen::MatrixXd::Zero( propagators::getSingleIntegrationSize( integratedStateType_ ), 1 );
114  }
115  return partial;

233  void getCurrentParameterPartial(
234  const boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter,
235  Eigen::Block< Eigen::MatrixXd > partialMatrix )
236  {
237  // Check if dependecy is computed
238  if( currentDoubleParameterPartials_.count( parameter ) == 0 )
239  {
240  // Check if dependecy exists at all
241  if( parameterDoublePartialFunctions_.count( parameter ) == 0 )
242  {
243  throw std::runtime_error(
244  "Parameter of type " +
245  boost::lexical_cast< std::string >( parameter->getParameterName( ).first ) + ", " +
246  parameter->getParameterName( ).second.first + ", " +
247  parameter->getParameterName( ).second.second + ", " +
248  " not found in list of existing partials" );
249  }
250  else
251  {
252  std::cerr<<"Warning, double partial should already be calculated"<<std::endl;
253  parameterDoublePartialFunctions_.at( parameter )( currentDoubleParameterPartials_[ parameter ] );
254  }
255  }
256  partialMatrix = currentDoubleParameterPartials_[ parameter ];
257  }

330  virtual int setParameterPartialUpdateFunction(
331  boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
332  {
333  // Get partial function.
334  std::pair< boost::function< void( Eigen::MatrixXd& ) >, int > parameterPartialFunction =
335  getParameterPartialFunction( parameter );
336 
337  // If partial function found, add function to list of computations to be performed when calling
338  // updateParameterPartials.
339  if( parameterPartialFunction.second > 0 && parameterDoublePartialFunctions_.count( parameter ) == 0 )
340  {
341  if( parameterDoublePartialFunctions_.count( parameter ) == 0 )
342  {
343  parameterDoublePartialFunctions_[ parameter ] = parameterPartialFunction.first;
344  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
345  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
346  }
347  }
348 
349  return parameterPartialFunction.second;
350  }

386  void updateParameterPartials( )
387  {
388  // Update member object partials.
389  updateParameterPartialsOfMemberObjects( );

391  // Update double parameter partials
392  for( parameterDoublePartialFunctionIterator_ = parameterDoublePartialFunctions_.begin( );

399  isCurrentDoubleParameterPartialSet_[ parameterDoublePartialFunctionIterator_->first ] = 1;
