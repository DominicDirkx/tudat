103  receiverState, transmitterState, time, isTimeAtReception );

106  Eigen::Matrix< ObservationScalarType, 3, 1 > sphericalRelativeCoordinates =
107  coordinate_conversions::convertCartesianToSpherical< StateScalarType >(
108  transmitterState.segment( 0, 3 ) - receiverState.segment( 0, 3 ) ).
