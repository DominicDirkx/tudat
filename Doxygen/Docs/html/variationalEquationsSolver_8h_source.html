264 template< typename TimeType, typename StateScalarType >
265 void setVariationalEquationsSolution(
266  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > >&
267  numericalIntegrationResult,
268  std::vector< std::map< double, Eigen::MatrixXd > >& variationalEquationsSolution,
269  const std::pair< int, int > stateTransitionStartIndices,
270  const std::pair< int, int > sensitivityStartIndices,
271  const int stateTransitionMatrixSize,
272  const int parameterSetSize )
273 {
274  variationalEquationsSolution.clear( );
275  variationalEquationsSolution.resize( 2 );
276 
277  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > >::iterator
278  integrationIterator = numericalIntegrationResult.begin( );
279  integrationIterator != numericalIntegrationResult.end( ); )
280  {
281  // Set result for state transition matrix in each time step.
282  variationalEquationsSolution[ 0 ][ integrationIterator->first ] =
283  ( integrationIterator->second.block( stateTransitionStartIndices.first, stateTransitionStartIndices.second,
284  stateTransitionMatrixSize,
285  stateTransitionMatrixSize ) ).template cast< double >( );
286 
287  // Set result for sensitivity matrix in each time step.
288  variationalEquationsSolution[ 1 ][ integrationIterator->first ] =
289  ( integrationIterator->second.block( sensitivityStartIndices.first, sensitivityStartIndices.second,
290  stateTransitionMatrixSize,
291  parameterSetSize -
292  stateTransitionMatrixSize ) ).template cast< double >( );
293  numericalIntegrationResult.erase( integrationIterator++ );
294  }
295 }

323 template< typename StateScalarType = double, typename TimeType = double, typename ParameterType = double >
324 bool checkPropagatorSettingsAndParameterEstimationConsistency(
325  const boost::shared_ptr< PropagatorSettings< StateScalarType > > propagatorSettings,
326  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate )
327 {
328  bool isInputConsistent = 1;
329 
330  // Check type of dynamics
331  switch( propagatorSettings->stateType_ )
332  {
333  case transational_state:
334  {
335  boost::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > > translationalPropagatorSettings =
336  boost::dynamic_pointer_cast< TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
337 
338  // Retrieve estimated and propagated translational states, and check equality.
339  std::vector< std::string > propagatedBodies = translationalPropagatorSettings->bodiesToIntegrate_;
340  std::vector< std::string > estimatedBodies = estimatable_parameters::getListOfBodiesWithTranslationalStateToEstimate(
341  parametersToEstimate );
342  if( propagatedBodies.size( ) != estimatedBodies.size( ) )
343  {
344  std::string errorMessage = "Error, propagated and estimated body vector sizes are inconsistent " +
345  boost::lexical_cast< std::string >( propagatedBodies.size( ) ) + " " +
346  boost::lexical_cast< std::string >( estimatedBodies.size( ) );
347  throw std::runtime_error( errorMessage );
348  isInputConsistent = 0;
349  }
350  else
351  {
352  for( unsigned int i = 0; i < propagatedBodies.size( ); i++ )
353  {
354  if( propagatedBodies.at( i ) != estimatedBodies.at( i ) )
355  {
356  std::string errorMessage = "Error, propagated and estimated body vectors inconsistent at index" +
357  boost::lexical_cast< std::string >( propagatedBodies.at( i ) ) + " " +
358  boost::lexical_cast< std::string >( estimatedBodies.at( i ) );
359  throw std::runtime_error( errorMessage );
360  isInputConsistent = 0;
361  }
362  }
363 
364  }
365  break;
366  }
367  default:
368  std::string errorMessage = "Error, cannot yet check consistency of propagator settings for type " +
369  boost::lexical_cast< std::string >( propagatorSettings->stateType_ );
370  throw std::runtime_error( errorMessage );
371  }
372  return isInputConsistent;
373 }

422  SingleArcVariationalEquationsSolver(
423  const simulation_setup::NamedBodyMap& bodyMap,
424  const boost::shared_ptr< numerical_integrators::IntegratorSettings< TimeType > > integratorSettings,
425  const boost::shared_ptr< PropagatorSettings< StateScalarType > > propagatorSettings,
426  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
427  const bool integrateDynamicalAndVariationalEquationsConcurrently = 1,
428  const boost::shared_ptr< numerical_integrators::IntegratorSettings< double > > variationalOnlyIntegratorSettings
429  = boost::shared_ptr< numerical_integrators::IntegratorSettings< double > >( ),
430  const bool clearNumericalSolution = 1,
431  const bool integrateEquationsOnCreation = 1 ):
432  VariationalEquationsSolver< StateScalarType, TimeType, ParameterType >(
433  bodyMap, integratorSettings, propagatorSettings, parametersToEstimate,
434  variationalOnlyIntegratorSettings, clearNumericalSolution )
435  {
436  // Check input consistency
437  if( !checkPropagatorSettingsAndParameterEstimationConsistency< StateScalarType, TimeType, ParameterType >(
438  propagatorSettings, parametersToEstimate ) )
439  {
440  throw std::runtime_error(
441  "Error when making single arc variational equations solver, estimated and propagated bodies are inconsistent" );
442  }
443  else
444  {
445  // Create simulation object for dynamics only.
446  dynamicsSimulator_ = boost::make_shared< SingleArcDynamicsSimulator< StateScalarType, TimeType > >(
447  bodyMap, integratorSettings, propagatorSettings, false, clearNumericalSolution, true );
448  dynamicsStateDerivative_ = dynamicsSimulator_->getDynamicsStateDerivative( );
449 
450  // Create state derivative partials
451  std::map< IntegratedStateType, orbit_determination::StateDerivativePartialsMap >
452  stateDerivativePartials =
453  simulation_setup::createStateDerivativePartials
454  < StateScalarType, TimeType, ParameterType >(
455  dynamicsStateDerivative_->getStateDerivativeModels( ), bodyMap, parametersToEstimate );
456 
457  // Create variational equations objects.
458  variationalEquationsObject_ = boost::make_shared< VariationalEquations >(
459  stateDerivativePartials, parametersToEstimate_,
460  dynamicsStateDerivative_->getStateTypeStartIndices( ) );
461  dynamicsStateDerivative_->addVariationalEquations( variationalEquationsObject_ );
462 
463  // Resize solution of variational equations to 2 (state transition and sensitivity matrices)
464  variationalEquationsSolution_.resize( 2 );
465 
466  // Integrate variational equations from initial state estimate.
467  if( integrateEquationsOnCreation )
468  {
469  if( integrateDynamicalAndVariationalEquationsConcurrently )
470  {
471  integrateVariationalAndDynamicalEquations( propagatorSettings->getInitialStates( ), 1 );
472  }
473  else
474  {
475  integrateVariationalAndDynamicalEquations( propagatorSettings->getInitialStates( ), 0 );
476  }
477  }
478  }

508  void integrateVariationalAndDynamicalEquations(
509  const VectorType& initialStateEstimate, const bool integrateEquationsConcurrently )
510  {
511  variationalEquationsSolution_[ 0 ].clear( );
512  variationalEquationsSolution_[ 1 ].clear( );
513 
514 
515  if( integrateEquationsConcurrently )
516  {
517 
518  // Create initial conditions from new estimate.
519  MatrixType initialVariationalState = this->createInitialConditions(
520  dynamicsStateDerivative_->convertFromOutputSolution(
521  initialStateEstimate, integratorSettings_->initialTime_ ) );
522 
523 
524  // Integrate variational and state equations.
525  dynamicsStateDerivative_->setPropagationSettings( std::vector< IntegratedStateType >( ), 1, 1 );
526  std::map< TimeType, Eigen::VectorXd > dependentVariableHistory;
527  std::map< TimeType, MatrixType > rawNumericalSolution;
528  EquationIntegrationInterface< MatrixType, TimeType >::integrateEquations(
529  dynamicsSimulator_->getStateDerivativeFunction( ), rawNumericalSolution,
530  initialVariationalState, integratorSettings_,
531  boost::bind( &PropagationTerminationCondition::checkStopCondition,
532  dynamicsSimulator_->getPropagationTerminationCondition( ), _1 ),
533  dependentVariableHistory );
534 
535  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > equationsOfMotionNumericalSolution;
536  utilities::createVectorBlockMatrixHistory(
537  rawNumericalSolution, equationsOfMotionNumericalSolution,
538  std::make_pair( 0, parameterVectorSize_ ), stateTransitionMatrixSize_ );
539 
540  equationsOfMotionNumericalSolution = convertNumericalStateSolutionsToOutputSolutions(
541  equationsOfMotionNumericalSolution, dynamicsStateDerivative_ );
542  dynamicsSimulator_->manuallySetAndProcessRawNumericalEquationsOfMotionSolution(
543  equationsOfMotionNumericalSolution );
544 
545  // Reset solution for state transition and sensitivity matrices.
546  setVariationalEquationsSolution< TimeType, StateScalarType >(
547  rawNumericalSolution, variationalEquationsSolution_,
548  std::make_pair( 0, 0 ), std::make_pair( 0, stateTransitionMatrixSize_ ),
549  stateTransitionMatrixSize_, parameterVectorSize_ );
550  }
551  else
552  {
553 
554  dynamicsStateDerivative_->setPropagationSettings( std::vector< IntegratedStateType >( ), 1, 0 );
555  dynamicsSimulator_->integrateEquationsOfMotion( initialStateEstimate );
556 
557  // Integrate variational equations.
558  dynamicsStateDerivative_->setPropagationSettings( boost::assign::list_of( transational_state ), 0, 1 );
559  Eigen::MatrixXd initialVariationalState = this->createInitialVariationalEquationsSolution( );
560  std::map< double, Eigen::MatrixXd > rawNumericalSolution;
561  std::map< TimeType, Eigen::VectorXd > dependentVariableHistory;
562 
563  EquationIntegrationInterface< Eigen::MatrixXd, double >::integrateEquations(
564  dynamicsSimulator_->getDoubleStateDerivativeFunction( ), rawNumericalSolution, initialVariationalState,
565  variationalOnlyIntegratorSettings_,
566  boost::bind( &PropagationTerminationCondition::checkStopCondition,
567  dynamicsSimulator_->getPropagationTerminationCondition( ), _1 ),
568  dependentVariableHistory );
569 
570  setVariationalEquationsSolution< double, double >(
571  rawNumericalSolution, variationalEquationsSolution_, std::make_pair( 0, 0 ),
572  std::make_pair( 0, stateTransitionMatrixSize_ ),
573  stateTransitionMatrixSize_, parameterVectorSize_ );
574 
575  }
576 
577  // Reset solution for state transition and sensitivity matrices.
578  resetVariationalEquationsInterpolators( );
579 
580  }
