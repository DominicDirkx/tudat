190  ObservationScalarType calculateLightTimeWithLinkEndsStates(
191  StateType& receiverStateOutput,
192  StateType& transmitterStateOutput,
193  const TimeType time,
194  const bool isTimeAtReception = 1,
195  const ObservationScalarType tolerance =
196  ( getDefaultLightTimeTolerance< ObservationScalarType, StateScalarType >( ) ) )
197  {
198  using physical_constants::SPEED_OF_LIGHT;
199  using std::fabs;
200 
201  // Initialize reception and transmission times and states to initial guess (zero light time)
202  TimeType receptionTime = time;
203  TimeType transmissionTime = time;
204  StateType receiverState = stateFunctionOfReceivingBody_( receptionTime );
205  StateType transmitterState =
206  stateFunctionOfTransmittingBody_( transmissionTime );
207 
208  // Set initial light-time correction.
209  setTotalLightTimeCorrection(
210  transmitterState, receiverState, transmissionTime, receptionTime );
211 
212  // Calculate light-time solution assuming infinte speed of signal as initial estimate.
213  ObservationScalarType previousLightTimeCalculation =
214  calculateNewLightTimeEstime( receiverState, transmitterState );
215 
216  // Set variables for iteration
217  ObservationScalarType newLightTimeCalculation = 0.0;
218  bool isToleranceReached = false;
219 
220  // Recalculate light-time solution until tolerance is reached.
221  int counter = 0;
222 
223  // Set variable determining whether to update the light time each iteration.
224  bool updateLightTimeCorrections = false;
225  if( iterateCorrections_ )
226  {
227  updateLightTimeCorrections = true;
228  }
229 
230  // Iterate until tolerance reached.
231  while( !isToleranceReached )
232  {
233  // Update light-time corrections, if necessary.
234  if( updateLightTimeCorrections )
235  {
236  setTotalLightTimeCorrection(
237  transmitterState, receiverState, transmissionTime, receptionTime );
238  }
239 
240  // Update light-time estimate for this iteration.
241  if( isTimeAtReception )
242  {
243  receptionTime = time;
244  transmissionTime = time - previousLightTimeCalculation;
245  transmitterState = ( stateFunctionOfTransmittingBody_( transmissionTime ) );
246  }
247  else
248  {
249  receptionTime = time + previousLightTimeCalculation;
250  transmissionTime = time;
251  receiverState = ( stateFunctionOfReceivingBody_( receptionTime ) );
252  }
253  newLightTimeCalculation = calculateNewLightTimeEstime( receiverState, transmitterState );
254 
255  // Check for convergence.
256  if( fabs( newLightTimeCalculation - previousLightTimeCalculation ) < tolerance )
257  {
258  // If convergence reached, but light-time corrections not iterated,
259  // perform 1 more iteration to check for change in correction.
260  if( !updateLightTimeCorrections )
261  {
262  updateLightTimeCorrections = true;
263  }
264  else
265  {
266  isToleranceReached = true;
267  }
268  }
269  else
270  {
271  // Get out of infinite loop (for instance due to low accuracy state functions,
272  // to stringent tolerance or limit case for trop. corrections).
273  if( counter == 20 )
274  {
275  isToleranceReached = true;
276  std::string errorMessage =
277  "Warning, light time unconverged at level " +
278  boost::lexical_cast< std::string >(
279  fabs( newLightTimeCalculation - previousLightTimeCalculation ) ) +
280  "; current light-time corrections are: " +
281  boost::lexical_cast< std::string >( currentCorrection_ ) + " and input time was " +
282  boost::lexical_cast< std::string >( time );
283  throw std::runtime_error( errorMessage );
284  }
285 
286  // Update light time for new iteration.
287  previousLightTimeCalculation = newLightTimeCalculation;
288  }
289 
290  counter++;
291  }
292 
293  // Set output variables and return the light time.
294  receiverStateOutput = receiverState;
295  transmitterStateOutput = transmitterState;
296 
297  return newLightTimeCalculation;
298  }
