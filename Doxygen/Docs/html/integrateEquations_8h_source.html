49 template< typename StateType = Eigen::MatrixXd, typename TimeType = double, typename TimeStepType = TimeType >
50 void integrateEquationsFromIntegrator(
51  const boost::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > > integrator,
52  const TimeStepType initialTimeStep,
53  const boost::function< bool( const double ) > stopPropagationFunction,
54  std::map< TimeType, StateType >& solutionHistory,
55  std::map< TimeType, Eigen::VectorXd >& dependentVariableHistory,
56  const boost::function< Eigen::VectorXd( ) > dependentVariableFunction =
57  boost::function< Eigen::VectorXd( ) >( ),
58  const int saveFrequency = TUDAT_NAN,
59  const TimeType printInterval = TUDAT_NAN )
60 {
61 
62  // Get Initial state and time.
63  TimeType currentTime = integrator->getCurrentIndependentVariable( );
64  TimeType initialTime = currentTime;
65  StateType newState = integrator->getCurrentState( );
66 
67  // Initialization of numerical solutions for variational equations
68  solutionHistory.clear( );
69  solutionHistory[ currentTime ] = newState;
70  dependentVariableHistory.clear( );
71 
72 
73  if( !dependentVariableFunction.empty( ) )
74  {
75  integrator->getStateDerivativeFunction( )( currentTime, newState );
76  dependentVariableHistory[ currentTime ] = dependentVariableFunction( );
77  }
78 
79  // Set initial time step and total integration time.
80  TimeStepType timeStep = initialTimeStep;
81  TimeType previousTime = currentTime;
82 
83  int saveIndex = 0;
84 
85  bool breakPropagation = 0;
86  // Perform numerical integration steps until end time reached.
87  do
88  {
89  try
90  {
91  previousTime = currentTime;
92 
93  // Perform integration step.
94  newState = integrator->performIntegrationStep( timeStep );
95  currentTime = integrator->getCurrentIndependentVariable( );
96  timeStep = integrator->getNextStepSize( );
97 
98  // Save integration result in map
99  saveIndex++;
100  saveIndex = saveIndex % saveFrequency;
101  if( saveIndex == 0 )
102  {
103  solutionHistory[ currentTime ] = newState;
104 
105  if( !dependentVariableFunction.empty( ) )
106  {
107  integrator->getStateDerivativeFunction( )( currentTime, newState );
108  dependentVariableHistory[ currentTime ] = dependentVariableFunction( );
109  }
110  }
111 
112 
113  // Print solutions
114  if( printInterval == printInterval )
115  {
116  if( ( static_cast<int>( std::fabs( static_cast< double >( currentTime - initialTime ) ) ) %
117  static_cast< int >( printInterval ) ) <
118  ( static_cast< int >( std::fabs( static_cast< double >( previousTime - initialTime ) ) ) %
119  static_cast<int>( printInterval ) ) )
120  {
121  std::cout<<"Current time and state in integration: "<<std::setprecision( 10 )<<
122  timeStep<<" "<<currentTime<<" "<<newState.transpose( )<<std::endl;
123  }
124  }
125  }
126  catch( const std::exception &caughtException )
127  {
128  std::cerr << caughtException.what( )<<std::endl;
129  std::cerr<<"Error, propagation terminated at t=" + boost::lexical_cast< std::string >( currentTime ) +
130  ", returning propagation data up to current time"<<std::endl;
131  breakPropagation = 1;
132  }
133  }
134  while( !stopPropagationFunction( static_cast< double >( currentTime ) ) && !breakPropagation );
135 }
