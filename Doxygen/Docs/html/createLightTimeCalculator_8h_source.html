70 {

72 
73  boost::function< StateType( const TimeType& ) > linkEndCompleteEphemerisFunction;
74 
75  // Checking transmitter if a reference point is to be used
76  if( linkEndId.second != "" )
77  {
78  if( bodyWithLinkEnd->getGroundStationMap( ).count( linkEndId.second ) == 0 )
79  {
80  std::string errorMessage = "Error when making ephemeris function for " + linkEndId.first + ", " + linkEndId.second + ", station not found.";
81  throw std::runtime_error( errorMessage );
82  }
83 
84  // Retrieve function to calculate state of transmitter S/C
85  linkEndCompleteEphemerisFunction =
86  boost::bind( &ephemerides::Ephemeris::getTemplatedStateFromEphemeris< StateScalarType,TimeType >,
87  createReferencePointEphemeris< TimeType, StateScalarType >(
88  bodyWithLinkEnd, bodyWithLinkEnd->getRotationalEphemeris( ),
89  boost::bind( &ground_stations::GroundStation::getStateInPlanetFixedFrame
90  < StateScalarType, TimeType >,
91  bodyWithLinkEnd->getGroundStation( linkEndId.second ), _1 ) ), _1 );
92 
93  }
94  // Else, create state function for center of mass
95  else
96  {
97  // Create function to calculate state of transmitting ground station.
98  linkEndCompleteEphemerisFunction = boost::bind( &simulation_setup::Body::getStateInBaseFrameFromEphemeris< StateScalarType, TimeType >,
99  bodyWithLinkEnd, _1 );
100  }
