607  void setDependentOrientationCalculator(
608  const boost::shared_ptr< reference_frames::DependentOrientationCalculator > dependentOrientationCalculator )
609  {
610  // Check if object already exists
611  if( dependentOrientationCalculator_ != NULL )
612  {
613  // Try to create closure between new and existing objects (i.e ensure that they end up computing the same rotation
614  // in differen manenrs.
615  if( ( boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
616  dependentOrientationCalculator ) != NULL ) &&
617  ( boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
618  dependentOrientationCalculator_ ) == NULL ) )
619  {
620  reference_frames::setAerodynamicDependentOrientationCalculatorClosure(
621  dependentOrientationCalculator_,
622  boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
623  dependentOrientationCalculator ) );
624  }
625  else if( ( boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
626  dependentOrientationCalculator_ ) != NULL ) &&
627  ( boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
628  dependentOrientationCalculator ) == NULL ) )
629  {
630  reference_frames::setAerodynamicDependentOrientationCalculatorClosure(
631  dependentOrientationCalculator,
632  boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
633  dependentOrientationCalculator_ ) );
634  }
635  else
636  {
637  std::cerr<< "Warning, cannot reset dependentOrientationCalculator, incompatible object already exists" <<std::endl;
638  }
639  }
640  else
641  {
642  dependentOrientationCalculator_ = dependentOrientationCalculator;
