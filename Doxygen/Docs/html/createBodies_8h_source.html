97 template< typename StateScalarType = double, typename TimeType = double >
98 void setGlobalFrameBodyEphemerides( const NamedBodyMap& bodyMap,
99  const std::string& globalFrameOrigin,
100  const std::string& globalFrameOrientation )
101 {
102  using namespace tudat::simulation_setup;
103  std::string ephemerisFrameOrigin;
104  std::string ephemerisFrameOrientation;
105  std::string rotationModelFrame;
106 
107  // Iterate over all bodies
108  for( NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( );
109  bodyIterator != bodyMap.end( ); bodyIterator++ )
110  {
111  // Check id body contains an ephemeris
112  if( bodyIterator->second->getEphemeris( ) != NULL )
113  {
114  // Retrieve ephemeris origin
115  ephemerisFrameOrigin = bodyIterator->second->getEphemeris( )->getReferenceFrameOrigin( );
116 
117  // Check if ephemeris origin differs from global origin.
118  if( ephemerisFrameOrigin != globalFrameOrigin )
119  {
120  // Check if correction can be made
121  if( bodyMap.count( ephemerisFrameOrigin ) == 0 )
122  {
123  throw std::runtime_error(
124  "Error, body " + bodyIterator->first + " has ephemeris in frame " +
125  ephemerisFrameOrigin + ", but no conversion to frame " + globalFrameOrigin +
126  " can be made" );
127  }
128  else
129  {
130  boost::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > stateFunction =
131  boost::bind( &Body::getStateInBaseFrameFromEphemeris< StateScalarType, TimeType >,
132  bodyMap.at( ephemerisFrameOrigin ), _1 );
133 
134  boost::shared_ptr< BaseStateInterface > baseStateInterface = boost::make_shared< BaseStateInterfaceImplementation< double, double > >(
135  ephemerisFrameOrigin, stateFunction );
136 
137  bodyIterator->second->setEphemerisFrameToBaseFrame( baseStateInterface );
138  }
139  }
140 
141  // Retrieve ephemeris orientation
142  ephemerisFrameOrientation = bodyIterator->second->getEphemeris( )->getReferenceFrameOrientation( );
143 
144  // If two are not equal, throw error.
145  if( ephemerisFrameOrientation != globalFrameOrientation )
146  {
147  throw std::runtime_error(
148  "Error, ephemeris orientation of body " + bodyIterator->first
149  + " is not the same as global orientation " + ephemerisFrameOrientation
150  + ", " + globalFrameOrientation );
151  }
152 
153 
154  }
155 
156  // Check if body has rotational ephemeris.
157  if( bodyIterator->second->getRotationalEphemeris( ) != NULL )
158  {
159  // Check if rotational ephemeris base frame orienatation is equal to to global
160  // orientation.
161  rotationModelFrame = bodyIterator->second->getRotationalEphemeris( )
162  ->getBaseFrameOrientation( );
163 
164  // Throw error if two frames are not equal.
165  if( rotationModelFrame != globalFrameOrientation )
166  {
167  throw std::runtime_error(
168  "Error, rotation base orientation of body " + bodyIterator->first +
169  " is not the same as global orientation " + rotationModelFrame + ", " +
170  globalFrameOrientation );
171  }
172  }
173  }
174 
175 }
