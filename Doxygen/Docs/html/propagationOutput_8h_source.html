157 template< typename TimeType = double, typename StateScalarType = double >
158 boost::function< double( ) > getDoubleDependentVariableFunction(
159  const boost::shared_ptr< SingleDependentVariableSaveSettings > dependentVariableSettings,
160  const simulation_setup::NamedBodyMap& bodyMap,
161  const std::unordered_map< IntegratedStateType,

163  std::unordered_map< IntegratedStateType,

166  boost::function< double( ) > variableFunction;
167 
168  // Retrieve base information on dependent variable
169  PropagationDependentVariables dependentVariable = dependentVariableSettings->variableType_;
170  const std::string& bodyWithProperty = dependentVariableSettings->associatedBody_;
171  const std::string& secondaryBody = dependentVariableSettings->secondaryBody_;
172 
173  // Check dependent variable type and create function accordingly.
174  switch( dependentVariable )
175  {
176  case mach_number_dependent_variable:
177  {
178  if( bodyMap.at( bodyWithProperty )->getFlightConditions( ) == NULL )
179  {
180  std::string errorMessage = "Error, no flight conditions available when requesting Mach number output of " +
181  bodyWithProperty + "w.r.t." + secondaryBody;
182  throw std::runtime_error( errorMessage );
183  }
184 
185  boost::function< double( const double, const double ) > functionToEvaluate =
186  boost::bind( &aerodynamics::computeMachNumber, _1, _2 );
187 
188  // Retrieve functions for airspeed and speed of sound.
189  boost::function< double( ) > firstInput =
190  boost::bind( &aerodynamics::FlightConditions::getCurrentAirspeed,
191  bodyMap.at( bodyWithProperty )->getFlightConditions( ) );
192  boost::function< double( ) > secondInput =
193  boost::bind( &aerodynamics::FlightConditions::getCurrentSpeedOfSound,
194  bodyMap.at( bodyWithProperty )->getFlightConditions( ) );
195 
196 
197  variableFunction = boost::bind( &evaluateBivariateFunction< double, double >,
198  functionToEvaluate, firstInput, secondInput );
199  break;
200  }
201  case altitude_dependent_variable:
202  if( bodyMap.at( bodyWithProperty )->getFlightConditions( ) == NULL )
203  {
204  std::string errorMessage = "Error, no flight conditions available when requesting altitude output of " +
205  bodyWithProperty + "w.r.t." + secondaryBody;
206  throw std::runtime_error( errorMessage );
207  }
208  variableFunction = boost::bind( &aerodynamics::FlightConditions::getCurrentAltitude,
209  bodyMap.at( bodyWithProperty )->getFlightConditions( ) );
210  break;
211  case airspeed_dependent_variable:
212  if( bodyMap.at( bodyWithProperty )->getFlightConditions( ) == NULL )
213  {
214  std::string errorMessage = "Error, no flight conditions available when requesting airspeed output of " +
215  bodyWithProperty + "w.r.t." + secondaryBody;
216  throw std::runtime_error( errorMessage );
217  }
218  variableFunction = boost::bind( &aerodynamics::FlightConditions::getCurrentAirspeed,
219  bodyMap.at( bodyWithProperty )->getFlightConditions( ) );
220  break;
221  case local_density_dependent_variable:
222  if( bodyMap.at( bodyWithProperty )->getFlightConditions( ) == NULL )
223  {
224  std::string errorMessage = "Error, no flight conditions available when requesting density output of " +
225  bodyWithProperty + "w.r.t." + secondaryBody;
226  throw std::runtime_error( errorMessage );
227  }
228  variableFunction = boost::bind( &aerodynamics::FlightConditions::getCurrentDensity,
229  bodyMap.at( bodyWithProperty )->getFlightConditions( ) );
230  break;
231  case radiation_pressure_dependent_variable:

234  std::string errorMessage = "Error, no radiation pressure interfaces when requesting radiation pressure output of " +
235  bodyWithProperty + "w.r.t." + secondaryBody;
236  throw std::runtime_error( errorMessage );

238  variableFunction = boost::bind( &electro_magnetism::RadiationPressureInterface::getCurrentRadiationPressure,
239  bodyMap.at( bodyWithProperty )->getRadiationPressureInterfaces( ).at( secondaryBody ) );
240  break;
241  case relative_distance_dependent_variable:
242  {
243  // Retrieve functions for positions of two bodies.
244  boost::function< double( const Eigen::Vector3d&, const Eigen::Vector3d& ) > functionToEvaluate =
245  boost::bind( &linear_algebra::computeNormOfVectorDifference, _1, _2 );
246  boost::function< Eigen::Vector3d( ) > firstInput =
247  boost::bind( &simulation_setup::Body::getPosition, bodyMap.at( bodyWithProperty ) );
248  boost::function< Eigen::Vector3d( ) > secondInput =
249  boost::bind( &simulation_setup::Body::getPosition, bodyMap.at( secondaryBody ) );

251  variableFunction = boost::bind(
252  &evaluateBivariateReferenceFunction< double, Eigen::Vector3d >, functionToEvaluate, firstInput, secondInput );
253  break;
254  }
255  case relative_speed_dependent_variable:
256  {
257  // Retrieve functions for velicoty of two bodies.

259  boost::bind( &linear_algebra::computeNormOfVectorDifference, _1, _2 );
260  boost::function< Eigen::Vector3d( ) > firstInput =
261  boost::bind( &simulation_setup::Body::getVelocity, bodyMap.at( bodyWithProperty ) );
262  boost::function< Eigen::Vector3d( ) > secondInput =
263  boost::bind( &simulation_setup::Body::getVelocity, bodyMap.at( secondaryBody ) );

267 
268  break;
269  }
270  case single_acceleration_norm_dependent_variable:

274  boost::dynamic_pointer_cast< SingleAccelerationDependentVariableSaveSettings >(
275  dependentVariableSettings );

278  std::string errorMessage = "Error, inconsistent inout when creating dependent variable function of type single_acceleration_norm_dependent_variable";
279  throw std::runtime_error( errorMessage );

299  else

302  boost::bind( &basic_astrodynamics::AccelerationModel3d::getAcceleration,

308  }
309  case total_acceleration_norm_dependent_variable:
310  {
311  // Retrieve model responsible for computing accelerations of requested bodies.

321  case relative_body_aerodynamic_orientation_angle_variable:
