65  double evaluatePdf( const Eigen::VectorXd& independentVariables )
66  {
67  Eigen::VectorXd distanceFromMean = ( independentVariables - mean_ ) ;
68  Eigen::MatrixXd location = -0.5 * (
69  distanceFromMean.transpose( ) * inverseCovarianceMatrix_ * distanceFromMean ) ;
70 
71  return std::exp( location( 0, 0 ) ) / ( std::pow( 2.0 * mathematical_constants::PI, dimension_ / 2.0 ) *
72  std::sqrt( determinant_ ) );
73  }
