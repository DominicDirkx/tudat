72 template< typename StateScalarType >
73 std::map< propagators::EnvironmentModelsToUpdate,
74  std::vector< std::string > > createEnvironmentUpdaterSettings(
75  const boost::shared_ptr< PropagatorSettings< StateScalarType > > propagatorSettings,
76  const simulation_setup::NamedBodyMap& bodyMap )
77 {
78  std::map< propagators::EnvironmentModelsToUpdate,
79  std::vector< std::string > > environmentModelsToUpdate;
80 
81  // Check dynamics type
82  switch( propagatorSettings->stateType_ )
83  {
84  case hybrid:
85  {
86  // Cast to derived type
87  boost::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
88  boost::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
89 
90  // Iterate over all propagation settings in hybrid model
91  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > singleAccelerationUpdateNeeds;
92 
93  for( typename std::map< IntegratedStateType,
94  std::vector< boost::shared_ptr< PropagatorSettings< StateScalarType > > > >::const_iterator
95  typeIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
96  typeIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); typeIterator++ )
97  {
98  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
99  {
100  // Create current propagation model from settings list (must not be hybrid).
101  if( typeIterator->first != hybrid )
102  {
103  singleAccelerationUpdateNeeds = createEnvironmentUpdaterSettings< StateScalarType >(
104  typeIterator->second.at( i ), bodyMap );
105 
106  // Add single model environment model update settings to full list
107  checkValidityOfRequiredEnvironmentUpdates( singleAccelerationUpdateNeeds, bodyMap );
108  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
109  }
110  else
111  {
112  throw std::runtime_error(
113  "Error when making environment updater type list, cannot handle hybrid propagator inside hybrid propagator" );
114  }
115  }
116  }
117  break;
118  }
119  // Retrieve environment model settings for translational dynamics
120  case transational_state:
121  {
122  environmentModelsToUpdate = createTranslationalEquationsOfMotionEnvironmentUpdaterSettings(
123  boost::dynamic_pointer_cast<
124  TranslationalStatePropagatorSettings< StateScalarType > >(
125  propagatorSettings )->accelerationsMap_,
126  bodyMap );
127  break;
128  }
129  // Retrieve environment model settings for mass rate model
130  case body_mass_state:
131  {
132  environmentModelsToUpdate = createMassPropagationEnvironmentUpdaterSettings(
133  boost::dynamic_pointer_cast<
134  MassPropagatorSettings< StateScalarType > >(
135  propagatorSettings )->massRateModels_, bodyMap );
136  break;
137  }
138  case custom_state:
139  {
140  break;
141  }
142  default:
143  {
144  throw std::runtime_error( "Error, cannot create environment updates for type " +
145  boost::lexical_cast< std::string >( propagatorSettings->stateType_ ) );
146  }
147  }
148  return environmentModelsToUpdate;
149 
150 }

173 template< typename StateScalarType, typename TimeType >
174 boost::shared_ptr< propagators::EnvironmentUpdater< StateScalarType, TimeType > >
175 createEnvironmentUpdaterForDynamicalEquations(
176  const boost::shared_ptr< PropagatorSettings< StateScalarType > > propagatorSettings,
177  const simulation_setup::NamedBodyMap& bodyMap )
178 {
179  // Create environment update settings.
180  std::map< IntegratedStateType,
181  std::vector< std::pair< std::string, std::string > > >integratedTypeAndBodyList =
182  getIntegratedTypeAndBodyList< StateScalarType >( propagatorSettings );
183 
184  std::map< propagators::EnvironmentModelsToUpdate,
185  std::vector< std::string > > environmentModelsToUpdate =
186  createEnvironmentUpdaterSettings< StateScalarType >( propagatorSettings, bodyMap );
187 
188  // Create and return environment updater object.
189  return boost::make_shared< EnvironmentUpdater< StateScalarType, TimeType > >(
190  bodyMap, environmentModelsToUpdate, integratedTypeAndBodyList );
191 }
