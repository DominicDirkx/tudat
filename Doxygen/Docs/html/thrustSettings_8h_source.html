529 class AccelerationLimitedThrottleGuidance: public ThrustInputParameterGuidance
530 {
531 public:
532 
534 
545  AccelerationLimitedThrottleGuidance(
546  const NamedBodyMap& bodyMap,
547  const std::string nameOfBodyWithGuidance,
548  const std::string nameOfCentralBody,
549  const std::vector< propulsion::ThrustIndependentVariables > independentVariables,
550  const boost::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator,
551  const double maximumAcceleration ): ThrustInputParameterGuidance( 1, 0, true, 0 ),
552  bodyWithGuidance_( bodyMap.at( nameOfBodyWithGuidance ) ), thrustInterpolator_( thrustInterpolator ),
553  maximumAcceleration_( maximumAcceleration )
554  {
555  // Split independent variables into environmental/guidance.
556  int numberOfThrottles = 0;
557  std::vector< propulsion::ThrustIndependentVariables > guidanceFreeIndependentVariables;
558  for( unsigned int i = 0; i < independentVariables.size( ); i++ )
559  {
560  if( independentVariables.at( i ) == propulsion::throttle_dependent_thrust )
561  {
562  numberOfThrottles ++;
563  }
564  else if( independentVariables.at( i ) == propulsion::guidance_input_dependent_thrust )
565  {
566  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, guidance input not supported" );
567  }
568  else
569  {
570  guidanceFreeIndependentVariables.push_back( independentVariables.at( i ) );
571  }
572  }
573 
574  // Check input consistency.
575  if( numberOfThrottles > 1 )
576  {
577  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, multiple throttles detected" );
578  }
579 
580  // Create thrust input.
581  if( bodyWithGuidance_->getFlightConditions( ) == NULL && nameOfCentralBody != "" )
582  {
583  bodyWithGuidance_->setFlightConditions(
584  createFlightConditions( bodyWithGuidance_,
585  bodyMap.at( nameOfCentralBody ),
586  nameOfBodyWithGuidance,
587  nameOfCentralBody ) );
588  }
589  thrustInputFunctions_ = getPropulsionInputVariables(
590  bodyWithGuidance_, guidanceFreeIndependentVariables );
591 
592  currentThrustInput_.resize( thrustInputFunctions_.size( ) );
593  }
594 
596 
600  void updateGuidanceParameters( )
601  {
602  // Compute environmental input variables for thrust.
603  for( unsigned int i = 0; i < thrustInputFunctions_.size( ); i++ )
604  {
605  currentThrustInput_[ i ] = thrustInputFunctions_.at( i )( );
606  }
607 
608  // Compute maximum thrust
609  double currentThrust = thrustInterpolator_->interpolate( currentThrustInput_ );
610  double currentMass = bodyWithGuidance_->getBodyMass( );
611 
612  // Set throttle (to < 1 if necessary)
613  if( currentThrust / currentMass < maximumAcceleration_ )
614  {
615  currentThrustGuidanceParameters_[ 0 ] = 1.0;
616  }
617  else
618  {
619  currentThrustGuidanceParameters_[ 0 ] = maximumAcceleration_ * currentMass / currentThrust;
620  }
621  }
622 
623 private:
624 
626  boost::shared_ptr< simulation_setup::Body > bodyWithGuidance_;
627 
629  boost::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator_;
630 
632  double maximumAcceleration_;
633 
635  std::vector< boost::function< double( ) > > thrustInputFunctions_;
636 
638  std::vector< double > currentThrustInput_;
639 };
