92  const boost::shared_ptr< ObservationBias< 1 > > observationBiasCalculator = NULL )
93  {
94  using namespace observation_models;
95 
96  boost::shared_ptr< observation_models::ObservationModel<
97  1, ObservationScalarType, TimeType, StateScalarType > > observationModel;
98 
99  // Check type of observation model.
100  switch( observableType )
101  {
102  case oneWayRange:
103  {
104  // Check consistency input.
105  if( linkEnds.size( ) != 2 )

107  std::string errorMessage =
108  "Error when making 1 way range model, " +
109  boost::lexical_cast< std::string >( linkEnds.size( ) ) + " link ends found";
110  throw std::runtime_error( errorMessage );
111  }
112  if( linkEnds.count( receiver ) == 0 )
113  {
114  throw std::runtime_error( "Error when making 1 way range model, no receiver found" );
115  }
116  if( linkEnds.count( transmitter ) == 0 )
117  {
118  throw std::runtime_error( "Error when making 1 way range model, no transmitter found" );
119  }
120 
121  // Create observation model
122  observationModel = boost::make_shared< OneWayRangeObservationModel<
123  ObservationScalarType, TimeType, StateScalarType > >(
124  createLightTimeCalculator< ObservationScalarType, TimeType, StateScalarType >(
125  linkEnds.at( transmitter ), linkEnds.at( receiver ),
126  bodyMap, singleObservableCorrections ), observationBiasCalculator );
127 
128  break;
129  }
130  default:
131  std::string errorMessage = "Error, observable " + boost::lexical_cast< std::string >( observableType ) +
132  " not recognized when making size 1 observation model.";
133  throw std::runtime_error( errorMessage );
134  }
135  return observationModel;
136  }
